    #include <QObject>
    #include <QIODevice>
    #include <QStringList>

    class VCardComparator : public QObject
    {
    public: // Constructors and destructor

        VCardComparator(

            const QStringList& excludedFields);

        QStringList* mOutputLines;
    };
